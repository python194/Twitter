# Mở file và đọc nội dung
with open('api.txt', 'r') as file:
    lines = file.readlines()

# Tìm và trích xuất nội dung trong dấu nháy kép

for line in lines:
    if 'number' in line:

        number = line.split('"')[1]

# In giá trị của biến number
